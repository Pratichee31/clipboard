def findLongestRegex(x, y, z):
    n = len(x)
    # Find the last position where z[i] is different from both x[i] and y[i]
    k = -1
    for i in range(n):
        if z[i] != x[i] and z[i] != y[i]:
            k = i

    if k == -1:
        return "-1"

    ALL = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    FULL = "[" + ALL + "]"            # length 28
    # At index k we exclude z[k]
    exclude_char = z[k]
    without = "[" + "".join(c for c in ALL if c != exclude_char) + "]"  # length 27

    # Build answer: FULL everywhere, WITHOUT at index k (the last valid index)
    parts = []
    for i in range(n):
        parts.append(without if i == k else FULL)
    return "".join(parts)
